bw <- function(p){
  for (i in seq_along(p$layers)) {
    p$layers[[i]]$mapping$colour <- NULL
    p$layers[[i]]$mapping$fill <- NULL
    p$layers[[i]]$aes_params$colour <- "black"
    p$layers[[i]]$aes_params$fill <- "grey"
}
return(p)
}
